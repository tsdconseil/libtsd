Warning
For the internal sample type, it is highly recommanded to choose an integer type, because the way the filter is implemented can make the filter diverge with flotting point arithmetic.
Example for interpolation
// Upsample by 16
auto R = 16, N = 4;
// Input sampling frequency = 10 MHz / 16
auto fe = (10e6f) / R;
// Test signal frequency
auto f = 50e3;
// Creation of the CIC filter:
// - input / output type = float
// - internal registers = int
auto cic = filtre_cic<float,int>({R, N, 1}, 'u');
// Build a signal pure sine wave at 50 KHz
ArrayXf t = linspace(0, 1e-4, fe*1e-4); // 0.1 ms time slot
ArrayXf x = (2*pi*f*t).sin();
// Scale the signal by 2^15, because internal computing inside
// CIC filter will be done in integer format.
x *= 32768;
// Use the CIC filter to upsample by a ratio of R
ArrayXf xu = cic->step(x);

Example for decimation
// Input sampling frequency = 10 MHz
auto fe = 10e6;
// Decimate by 16 (e.g. output sample frequency = 625 KHz)
auto R = 16, N = 4;
// Creation of the CIC filter
auto cic = filtre_cic<float,int>({R, N});
// Build a signal which is the superposition of 2 pure sine wave:
// - one at 2 MHz
// - one at 20 KHz
auto f1 = 2e6/2, f2 = 20e3/2;
// 1 ms time slot
ArrayXf t = intervalle_temporel(/*10000*/2000, fe);
ArrayXf x = (t*2*tsd::pi*f1).sin() + (t*2*tsd::pi*f2).sin();
// Scale the data to 16 bit precision
x *= 32768;
// Decimation by ratio 1/R
ArrayXf xd = cic->step(x);
