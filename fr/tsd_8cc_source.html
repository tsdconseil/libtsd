300  void step(const Eigen::Ref<const Vecteur<T>> x)
301  {
302  if((windex == 0) && (x.rows() == N))
303  {
304  if(callback)
305  callback(x);
306  return;
307  }
308 
309  if((tampon.rows() == 0) && (N > 0))
310  tampon.resize(N);
311 
312  int nb_ech_entree = x.rows();
313  int i = 0;
314  while(i < nb_ech_entree)
315  {
316  /* Write index + number of bytes remaining in
317  * the input datablock. */
318  int new_windex = windex + nb_ech_entree - i;
319 
320  /* If too much data for the FIFO */
321  if(new_windex >= N)
322  new_windex = N;
323 
324  /* Copy input data to the FIFO */
325  int nech = new_windex - windex;
326 
327  tampon.segment(windex, nech) = x.segment(i, nech);
328 
329  i += nech;
330  windex = new_windex;
331 
332  if(windex == N)
333  {
334  if(callback)
335  callback(tampon);
336  windex = 0;
337  }
338  }
339  }
