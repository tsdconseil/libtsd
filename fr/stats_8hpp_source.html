111  auto r_vers_R(const Eigen::ArrayBase<derived> &r)
112  {
113  using T = typename derived::Scalar;
114  using rtype = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>;
115  int n = r.rows();
116  rtype R(n, n);
117  for(auto i = 0; i < n; i++)
118  {
119  for(auto j = 0; j < n; j++)
120  {
121  R(i,j) = r(std::abs(i-j));
122  if constexpr(est_complexe<T>())
123  if(j > i)
124  R(i,j) = conj(R(i,j));
125  }
126  }
127  return R;
128  }
