124  auto r_vers_R(const Eigen::ArrayBase<derived> &r)
125  {
126  using T = typename derived::Scalar;
127  using rtype = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>;
128  int n = r.rows();
129  rtype R(n, n);
130  for(auto i = 0; i < n; i++)
131  {
132  for(auto j = 0; j < n; j++)
133  {
134  R(i,j) = r(std::abs(i-j));
135  if constexpr(est_complexe<T>())
136  if(j > i)
137  R(i,j) = conj(R(i,j));
138  }
139  }
140  return R;
141  }
