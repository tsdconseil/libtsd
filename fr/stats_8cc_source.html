15  ArrayXf levinson_reel(const ArrayXf &r)
16  {
17  int n = r.rows();
18  ArrayXf a(n);
19  // R[i] - autocorrelation coefficients
20  // A[i] - filter coefficients
21  // K - reflection coefficients
22  // Alpha - prediction gain
23  a(0) = 1;
24  float k = - r(1) / r(0);
25  a(1) = k;
26  float α = r(0) * (1-k*k);
27 
28  for(auto i = 2; i <= n - 1; i++)
29  {
30  float s = (r.segment(1,i-1) * a.segment(1, i-1).reverse()).sum() + r(i);
31  k = -s / α;
32  a.segment(1, i-1) += k * a.segment(1, i-1).reverse().eval();
33  a(i) = k;
34  α *= (1-k*k);
35  }
36  return a;
37  }

42  ArrayXf levinson(const ArrayXf &l1, const ArrayXf &c1, const ArrayXf &y)
43  {
44  int n = l1.rows();
45 
46  //msg("y = {}", y.transpose());
47 
48  tsd_assert_msg(n == c1.rows(), "Levinson : ligne et colonne doivent être de dimensions identiques.");
49  tsd_assert_msg(n == y.rows(), "Levinson : dimensions invalides.");
50  tsd_assert_msg(l1(0) == c1(0), "Levinson : vecteurs invalides.");
51 
52  // t(n-1), t(n-2), ... , t(1) (t(0) exclu)
53  Eigen::VectorXf c1r = c1.tail(n-1).reverse().matrix();
54  // t(-1), t(-2), ..., t(-(n-1))
55  Eigen::VectorXf l1r = l1.tail(n-1).matrix();
56 
57  // Calcul des vecteurs avant et arrière
58  // et de la solution
59  Eigen::VectorXf f(n), b(n);
60  f(0) = 1 / l1(0);
61  b(0) = 1 / c1(0);
62 
63  Eigen::VectorXf x = Eigen::VectorXf::Zero(n);
64  x(0) = y(0) / l1(0);
65 
66  // i = dimension nouveau vecteur
67  for(auto i = 2; i <= n; i++)
68  {
69 
70  Eigen::VectorXf fb(i), bb(i);
71  fb.head(i-1) = f.head(i-1);
72  fb(i-1) = 0;
73  bb(0) = 0;
74  bb.tail(i-1) = b.head(i-1);
75 
76  //float ef = M.row(i).head(i-1).dot(f);
77  //float eb = M.row(0).segment(1,i-1).dot(b);
78 
79  float ef = c1r.tail(i-1).dot(f.head(i-1));
80  float eb = l1r.head(i-1).dot(b.head(i-1));
81  float fr = 1/(1 - eb * ef);
82 
83  // Mise à jour forward / backward
84  f.head(i) = fr * fb - ef * fr * bb;
85  b.head(i) = fr * bb - eb * fr * fb;
86 
87  float ex = c1r.tail(i-1).dot(x.head(i-1));
88 
89  x.head(i) += (y(i-1) - ex) * b.head(i);
90 
91 # if 0
92  // Check des vecteurs f / b
93  MatrixXf Ri(i, i);
94  for(auto k = 0; k < i; k++)
95  {
96  for(auto l = 0; l < i; l++)
97  {
98  if(k >= l)
99  Ri(k, l) = c1(k-l);
100  else
101  Ri(k, l) = l1(l-k);
102  }
103  }
104  Eigen::VectorXf Rf = Ri * f.head(i);
105  Eigen::VectorXf Rb = Ri * b.head(i);
106  Eigen::VectorXf Rx = Ri * x.head(i);
107  msg("Rf[{}] = {}", i, Rf.transpose());
108  msg("Rb[{}] = {}", i, Rb.transpose());
109  msg("Rx[{}] = {}", i, Rx.transpose());
110 # endif
111  }

113  return x.array();
114  }

145  SubSpaceSpectrum subspace_spectrum(const MatrixXcf &R, const SubSpaceSpectrumConfig &config)
146  {
147  int Ns = config.Ns;
148  Eigen::SelfAdjointEigenSolver<MatrixXcf> eig(R.matrix());
149 
150  int m = R.rows(); // Nombre d'antennes en DOA
151 
152  if(Ns == -1)
153  {
154  // Dans l'ordre de valeur absolue croissante
155  ArrayXcf l = eig.eigenvalues().array().reverse();
156  ArrayXf la = l.abs();
157 
158  ArrayXf vmdl = ArrayXf::Zero(m-1);
159 
160  // Calcul du critère MDL pour déterminer le nombre de sources
161  // D'après "Williams, D.B. “Detection: Determining the Number of Sources”
162  // Digital Signal Processing Handbook
163  for(auto Ns = 1; Ns < m; Ns++)
164  {
165  // N est le nombre d'observations en DOA
166  int N = m;//n; // à confirmer
167  // Produit des valeurs propres dans le bruit
168  // (attention : tends vers zéro très vite ?)
169  // -> Calcul de la somme des logs
170  float somme = la.tail(m - Ns).log().sum();
171  float mdl = -N * (somme - (Ns - m) * (-log(m-Ns) + somme)) + 0.5 * Ns * (2*m-Ns) * log(N);
172 
173  vmdl(Ns-1) = mdl;
174  }
175 
176  //msg("mdl = {}", vmdl);
177 
178  vmdl.maxCoeff(&Ns);
179  Ns++;
180 
181  if(config.debug_actif)
182  {
183  Figures f;
184  f.subplot().plot(10 * la.log10(), "", "Valeurs propres (log)");
185  auto s = f.subplot();
186  s.plot(linspace(1, m-1, m-1), vmdl);
187  s.titres("Critère MDL", "Nombre de sources", "MDL");
188  //f.enregistrer("./res/st/mdl.png");
189  f.afficher("MDL");
190  }
191  }
192 
193  tsd_assert_msg(Ns < m, "Le nombre de sources doit être inférieur à la dimension de la matrice de covariance (ici Ns={}, m={}).", Ns, m);
194

196  //Nsm = nfreqs;

206  for(auto i = 0; i < config.Nf; i++)
207  {
208  // Calcul du "vecteur de steering" (ici simple exponentielle complexe)
