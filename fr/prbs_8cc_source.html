22 // Exemples de polynômes primitif, encodés en binaire
23 // d'après "On Pseudo-Random and Orthogonal Binary Spreading Sequences", Mitra, 2007
24 // (bit de poids faible implicite = x^n,
25 // bit de poids fort implicite = x^0)
26 const std::vector<uint32_t> pols_prim =
27  {0, // n=0, n/a
28  0, // n=1, 1 + x
29  (1 << 1), // n=2, 1 + x + x^2
30  (1 << 2), // n=3, 1 + x + x^3
31  (1 << 3), // n=4, 1 + x + x^4
32  (1 << 3), // n=5, 1 + x^2 + x^5
33  (1 << 5), // n=6, 1 + x + x^6
34  (1 << 6), // n=7, 1 + x + x^7
35  (1 << 6) | (1 << 5) | (1 << 4), // n=8, 1 + x^2 + x^3 + x^4 + x^8
36  (1 << 5), // n=9
37  (1 << 7), // n=10
38  (1 << 9), // n=11
39  (1 << 9) | (1 << 8) | (1 << 5), // n=12, 1+x^3+x^4+x^7+x^12
40  (1 << 9) | (1 << 10) | (1 << 12), // n=13, 1+x+x^3+x^4+x^13
41  (1 << 5) | (1 << 3) | (1 << 1),
42  (1 << 14),
43  (1 << 5) | (1 << 3) | (1 << 2)
44 };

53 Poly<int> polynome_primitif(int n)
54 {
55  Poly<int> res;
56  Poly<int> z = Poly<int>::z;
57  auto p = polynome_primitif_binaire(n);
58  for(auto i = 0; i < n; i++)
59  {
60  if(p & (1 << i))
61  res = res + (z ^ (n-i));
62  }
63  res = res + Poly<int>::one(); // Ajoute c0 = 1, non utilisé dans les calculs
64  return res;
65 }

167 struct LFSRRecepteur::Impl
168 {
169  uint16_t reglen = 0;
170  uint32_t reg = 0, pol = 0;
171  enum
172  {
173  PRBS_STATE_UNLOCKED = 0,
174  PRBS_STATE_LOCKED
175  } state = PRBS_STATE_UNLOCKED;
176  uint32_t nb_consecutive_bits_errors = 0;
177  uint32_t nb_consecutive_bits_ok = 0;
178  uint64_t bit_counter = 0;
179  uint32_t nb_bits_a_ignorer = 0;
180  bool est_verouille = false;
181  // Number of bits decoded (right or bad) since the last change in locked state
182  uint64_t nb_bits = 0;
183  // Number of errors detected since the last change in locked state
184  uint64_t nb_erreurs = 0;
185 
186  void step(const BitStream &bs)
187  {
188  auto nbits = bs.lon();
189  for(auto i = 0; i < nbits; i++)
190  {
191  uint8_t digital_bit = bs[i];
192  //uint8_t digital_bit = (v[i] >> (7-j)) & 1;
193 
194  auto tmp = reg & pol;
195  auto sum = 0u;
196  for(auto k = 0u; k < reglen; k++)
197  sum = sum ^ ((tmp >> k) & 1);
198 
199  /* Unlocked state */
200  if(state == Impl::PRBS_STATE_UNLOCKED)
201  {
202  /* Fill register with received data */
203  reg = (reg >> 1) | (digital_bit << (reglen - 1));
204 
205  if(digital_bit == sum)
206  {
207  /* Bit received == processed: no error */
208  nb_consecutive_bits_ok++;
209  }
210  else
211  {
212  /* Bit received != processed: error */
213  nb_consecutive_bits_ok = 0;
214  }
215  if(nb_consecutive_bits_ok > 20)
216  {
217  //printf("prbs lock @%Ld.\n", bit_counter);
218  /* 8 consecutive bits are good: lock the receiver */
219  state = PRBS_STATE_LOCKED;
220  nb_consecutive_bits_errors = 0;
221  //nb_bits = 0;
222  //nb_errors = 0;
223  est_verouille = true;
224  }
225  }
226  /* Locked state */
227  else
228  {
229  /* Fill register with computed data */
230  reg = (reg >> 1) | (sum << (reglen - 1));
231  nb_bits++;
232 
233  if(digital_bit == sum)
234  {
235  /* Bit received == processed: no error */
236  nb_consecutive_bits_errors = 0;
237  }
238  else
239  {
240  /* Bit received != processed: error */
241  nb_consecutive_bits_errors++;
242  nb_erreurs++;
243  //printf("Error @bit %Ld.\n", bit_counter);
244  }
245  if(nb_consecutive_bits_errors > 5)
246  {
247  //printf("prbs unlock @%Ld after %Ld bits (%Ld errors).\n",
248  //bit_counter, nb_bits, nb_errors);
249  /* 5 consecutive bits are bad: unlock the receiver */
250  state = PRBS_STATE_UNLOCKED;
251  nb_consecutive_bits_ok = 0;
252  }
253  }
254  bit_counter++;
255  } // for(i) : bits
256  }
257 };
