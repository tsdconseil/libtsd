360  void step1(const ArrayXf &x)
361  {
362  int N = cqtk.N;
363 
364  tsd_assert(x.rows() == nb_data_attendu);
365 
366  // (1) Data rotation
367  // to remove to keep
368  // |----------------|---------|
369  // n_waited n_keep
370  cirbuffer.head(N - nb_data_attendu) = cirbuffer.tail(N - nb_data_attendu);
371  cirbuffer.tail(nb_data_attendu) = x;
372 
373  // (2) Change to frequency domain
374  //ArrayXcf X = fft(cirbuffer);
375 
376  // TODO : utiliser plan rfft
377  ArrayXcf x2 = cirbuffer;
378  ArrayXcf X;
379  plan_fft->step(x2, X);
380 
381  int minimal_position = N;
382 
383 
384  // (3) Kernels correlations
385  for(auto i = 0; i < cqtk.nfreqs; i++)
386  {
387  int ksize = cqtk.ksize(i);
388  int kmin = cqtk.kimin(i);
389  int ktime = cqtk.ktime(i);
390  int kpos = kposition(i);
391  int kapos = kaposition(i);
392 
393  ArrayXcf kerncx = cqtk.noyaux.col(i).head(ksize) * (X.segment(kmin, ksize)).conjugate();
394  int dt = (int) floor(ktime / 2);
395 
396  while(kpos + ktime <= N)
397  {
398  // Time shift, in samples, from the middle of the frame
399  float shift = (-kpos + 0.5*ktime) / N - 0.5f;
400 
401  // OL
402  cfloat ol0 = std::polar(1.0f, 2 * π_f * shift);
403  // Mixer for + freqs
404  cfloat s0 = std::polar(1.0f, 2 * π_f * kmin * shift);
405 
406  // Trop consommateur
407  //float cor = std::abs((tsd::polar(shift*posindex) * kerncx).sum());
408  cdouble correl = 0;
409  for(auto j = 0; j < ksize; j++)
410  {
411  /* shift kernel in time domain */
412  correl += kerncx(j) * s0;
413  /* Update mixer */
414  s0 *= ol0;
415  }
416 
417  sortie.push_back({i, (float) kapos, (float) std::abs(correl)});
418 
419  // Shift kernel (with overlap 1/2)
420  kpos += dt;
421  kapos += dt;
422  } // end while pos
423  if(kpos < minimal_position)
424  minimal_position = kpos;
425  kposition(i) = kpos;
426  kaposition(i) = kapos;
427  } // end for i=1::nfreqs
428 
429  // Nb data waited = minimal_position
430  nb_data_attendu = minimal_position;
431  kposition -= minimal_position;
432  }

434  std::tuple<ArrayXf, ArrayXf, ArrayXXf> interpolation(float ofs)
435  {
436  // Nombre total d'échantillons (irréguliers)
437  int ns = sortie.size();
438  // Nombre d'échantillons d'entrée / échantillon de sortie
439  float oprd = cqtk.fs / ofs;
440 
441  msg("interpolation : ofs = {} Hz, nptr = {}.", ofs, sortie.size());
442 
443  ArrayXf lvalue = ArrayXf::Zero(cqtk.nfreqs);
444  ArrayXf lindex = lvalue;
445 
446  // Index d'écriture, pour chaque fréquence
447  ArrayXi offset = ArrayXi::Zero(cqtk.nfreqs);
448 
449  // Durée totale de l'analyse (en secondes)
450  float duree = sortie[ns-1].temps / cqtk.fs;
451 
452  // Nombre d'échantillons à sortie (par fréquence)
453  int onmax = (int) ceil(duree * ofs);
454 
455  msg("CQT interpolation : duration = {} samples = {} seconds.", sortie[ns-1].temps, duree);
456  msg("CQT interpolation : creation of a matrix of size {} * {}.", onmax, cqtk.nfreqs);
457 
458  ArrayXXf A = ArrayXXf::Zero(onmax, cqtk.nfreqs);
459  // Echantillons classés par indices de fréquence croissants
460  for(auto i = 0; i < ns; i++)
461  {
462  auto &spl = sortie[i];
463  int fr = spl.id_freq;
464  if(fr >= cqtk.nfreqs)
465  echec("ofreq invalid.");
466 
467  // Temps échantillon précédent (pour cette fréquence), et échantillon courant
468  // (en nombre d'échantillons d'entrée)
469  float t0 = lindex(fr), t1 = spl.temps;
470  // Valeur précédente et courante
471  float y0 = lvalue(fr), y1 = std::abs(spl.valeur);
472  // Compléte la matrice de sortie
473  while(offset(fr) * oprd <= t1)
474  {
475  float t = offset(fr) * oprd;
476  float val;
477  if(t1 == t0)
478  val = y1;
479  else
480  val = y0 + ((t - t0) / (t1 - t0)) * (y1 - y0);
481  A(offset(fr),fr) = val;
482  offset(fr)++;
483  }
484  lindex(fr) = t1;
485  lvalue(fr) = y1;
486  }
487  ArrayXf t = linspace(0,A.rows()/ofs,A.rows());
488  // [t,f,A]
489  return {t, cqtk.freqs, A};
490  }
