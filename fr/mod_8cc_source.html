137  ArrayXcf step(const BitStream &bs)
138  {
139  // TODO : utiliser config.wf->génère_échantillons
140  auto x = wf->génère_symboles(bs);
141  auto x_symb = x;
142 
143  // Filtre de mise en forme, avec sur-échantillonnage intégré
144  x = filtre_mise_en_forme->step(x);
145 
146  // Fait maintenant directement dans le filtre
147  /*if(config.wf->filtre.type == SpecFiltreMiseEnForme::GAUSSIEN)
148  {
149  auto ma = tsd::filtrage::filtre_mg<cfloat,cdouble>(osf);
150  x = ma->step(x);
151  }*/
152 
153  ArrayXcf x_filtre = x;
154 
155  ArrayXf vfreqs, vphase;
156 
157  if(config.wf->est_fsk)
158  {
159  // df = 0.5 * h * fsymb
160  // df sur un symbole = 0.5 * h / osf
161  // 2 π / osf <=> h = 2
162  // => θ = h * 2 * pi / osf / 2 = h * pi / osf
163 
164  auto Ω_max = (π * config.wf->index) / osf;
165  // h = 2 -> Omega_max = 2 * pi / osf
166 
167  //msg("Ω max = {} degrés.", rad2deg(Ω_max));
168 
169  // h = 2 fd / fsymb = excursion / fsymb
170  // => omega_max = 2%pi*fd = %pi * h * fsymb
171  // => Omega_max = omega_max / fs
172  // = %pi * h / ovs;
173  // Conversion phase -> IQ
174 
175  ArrayXf xr = x.real();
176  // normalisation entre [-fmax,fmax]
177  vfreqs = xr * (Ω_max / xr.abs().maxCoeff());
178 
179  // TODO : échantillon précédent !
180  vphase = cumsum(vfreqs);
181  x = polar(vphase);
182  }
183 
184  if(fe1 != config.fe)
185  {
186  x = ra->step(x);
187  }
188 
189  // Modulation RF
190  if(config.fi != 0)
191  {
192  x *= ol->step(x.rows());
193  if(config.sortie_reelle)
194  x = x.real();
195  }
196 
197 
198  if(config.debug_actif)
199  {
200  msg("<h2>Modulation</h2>");
201 
202  {
203  Figures f;

209

211 
212  f.subplot().plot_iq(x_filtre, "ob-", "Constellation filtree");
213 
214  f.afficher("Modulation (1)");
215  }

226  f.subplot().plot(vphase, "", "Phase (cumsum)");
