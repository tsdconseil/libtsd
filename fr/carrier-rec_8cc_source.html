44 struct LF2: LoopFilter
45 {
46  float A, BL, η, γ, ρ, θ, μ, last_ped;
47 
48  LF2(float BL, float η)
49  {
50  this->BL = BL;
51  this->η = η;
52  A = 1; // PED gain at origin = 1 (ped supposed to be normalized)
53  γ = (16 * η * η * BL) / (A * (1 + 4 * η * η));
54  ρ = (4 * BL) / (1 + 4 * η * η);
55  reset();
56  }
57  void reset()
58  {
59  θ = μ = last_ped = 0;
60  }
61  float step(float x)
62  {
63  θ += μ;
64  μ += γ * ((1 + ρ) * x - last_ped);
65  last_ped = x;
66  return θ;
67  }
68 };

102 Ped ped_costa(int M)
103 {
104  if((M != 2) && (M != 4))
105  {
106  msg_erreur("ped_costa: fonctionne seulement en BPSK (M=2) ou QPSK (M=4), ici M = {}.", M);
107  return Ped();
108  }
109  // QPSK : m^4 = 1
110  if(M == 2)
111  // BPSK : m² = 1
112  // sin(phi) * cos(phi) = 0.5 sin(2phi) ~ phi
113  // Note: c'est la même chose qu'une squaring loop
114  // en prenant slmt la partie imaginaire.
115  return [](cfloat x) {return x.real() * x.imag();};
116  else if(M == 4)
117  {
118  // QPSK costa loop locks to square constellation
119  // And we expect a "losange" constellation.
120  return [](cfloat x){
121  cfloat z = x * std::polar(1.0f, π_f/4);
122  return z.imag() * sign(z.real()) - z.real() * sign(z.imag());
123  };
124  }
125  return Ped();
126 }

447 void SuiviPicFrequence::step(const ArrayXcf &x, float &freq_detectee, float &snr)
448 {
449  ArrayXcf X;
450  fft_plan->step(x, X, true);
451 
452  int i2 = 0;
453  auto a2 = X.abs2();
454  float y2 = a2.maxCoeff(&i2);
455 
456  snr = y2 / a2.mean();
457 
458  // Interpolation barycentrique (https://dspguru.com/dsp/howtos/how-to-interpolate-fft-peak/)
459  int i1 = (i2 - 1 + N) % N;
460  int i3 = (i2 + 1) % N;
461 
462  auto y1 = a2(i1);
463  auto y3 = a2(i3);
464 
465  // i = N/2 <=> f = fs/2
466  // => i <=> i * fs / N
467  if(i2 >= (int) N/2)
468  i2 = i2 - N; // 0.6 => -0.4
469 
470  freq_detectee = ((float) i2) / N;
471  tsd_assert(std::abs(freq_detectee) <= 0.5);
472 
473  auto d = (y3 - y1) / (y1 + y2 + y3 + 1e-30f);
474 
475  tsd_assert(std::abs(d) <= 1);
476 
477  freq_detectee += d / N;
478 }
