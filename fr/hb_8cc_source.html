84  void step(const Eigen::Ref<const Vecteur<T>> x, Vecteur<T> &y)
85  {
86  tsd_assert(K > 0);
87  int n = x.rows();
88  auto iptr = x.data();
89  auto optr = y.data();
90  if(iptr != optr)
91  {
92  y.resize((n + odd) / R);
93  optr = y.data();
94  }
95 
96  for(auto j = 0; j < n; j++)
97  {
98  auto cptr = coefs.data();
99 
100  T somme = 0;
101 
102  // La moitié des coefficients sont nuls.
103  // .... x3 x2 x1 x0
104  // -> y0 = x0 * h0 + x2 * h1 + ...
105  // -> y1 = x2 * h0 + ?
106 
107  // Donc on n'utilise que le signal pair ?
108  // xpair -> H0
109  // ximpair -> H1 (nul)
110 
111  // Attention, 3 coefficients centrals non nuls puis zéros réguliérement.
112  // Disons filtre de type h0 0 h1 0 h2
113  //
114 
115  // Remplace les deux plus ancien éléments
116  fenêtre(index) = *iptr++;
117  index = (index + 1) % K;
118 
119  // Ne calcule qu'un échantillon sur R
120  if(odd < R - 1)
121  {
122  odd++;
123  continue;
124  }
125  odd = 0;
126 
127 
128  T *wptr = fenêtre.data() + index;
129 
130  // Nombre d'échantillons à la fin de la ligne à retard
131  int K1 = K - index;
132  // Nombre d'échantillons au début de la ligne à retard
133  int K2 = K - K1;
134 
135  int i;
136  for(i = 0; i < K1; i += 2) // 1 coefficient sur 2 du filtre est nul
137  {
138  somme += *wptr * *cptr;
139  wptr += 2;
140  cptr += 2;
141  }
142 
143  i = i - K1;
144 
145  // Redémarre au début de la ligne à retard
146  wptr = fenêtre.data() + i;
147  for(; i < K2; i += 2)
148  {
149  somme += *wptr * *cptr;
150  wptr += 2;
151  cptr += 2;
152  }
153 
154  // Coefficient central non nul = 0.5 (forcément)
155  somme += /*coefs(K/2)*/ 0.5f * fenêtre((index + K/2) % K);
156 
157 
158  /*
159  *
160  * Pour i = 0; i < N; i += R
161  * (fenêtre(index + i [N]) + fenêtre(index - 1 - i [N])) * coefs(i)
162  *
163  *
164  *
165  */
166 
167  *optr++ = somme;
168  }
169  }

276 struct FiltreRIFUps: FiltreGen<T>
277 {
278  Vecteur<T> fenêtre;
279  int index, K;
280  Vecteur<Tc> coefs;
281  int odd = 0, R = 0;
282 
283 
284  FiltreRIFUps(const Vecteur<Tc> &c, int R)
285  {
286  this->R = R;
287  // Afin de préserver l'amplitude du signal
288  this->coefs = c * R;
289  odd = 0;
290  index = 0;
291  K = coefs.rows();
292 
293  // Complète avec des zéros
294  if((K % R) != 0)
295  {
296  coefs = vconcat(coefs, Vecteur<Tc>::Zero(R - (K % R)));
297  K = coefs.rows();
298  }
299 
300  tsd_assert((K % R) == 0);
301 
302  fenêtre.setZero(K/R);
303  }
304 
305  void step(const Eigen::Ref<const Vecteur<T>> x, Vecteur<T> &y)
306  {
307  tsd_assert(K > 0);
308  int n = x.rows();
309  auto iptr = x.data();
310  auto optr = y.data();
311  if(iptr != optr)
312  {
313  y.resize(n*R);
314  optr = y.data();
315  }
316 
317  for(auto j = 0; j < n; j++)
318  {
319 
320  // La moitié des coefficients sont nuls.
321  // .... x3 x2 x1 x0
322  // -> y0 = x0 * h0 + x2 * h1 + ...
323  // -> y1 = x2 * h0 + ?
324 
325  // Donc on n'utilise que le signal pair ?
326  // xpair -> H0
327  // ximpair -> H1 (nul)
328 
329  // Attention, 3 coefficients centrals non nuls puis zéros réguliérement.
330  // Disons filtre de type h0 0 h1 0 h2
331  //
332 
333  // Remplace les deux plus ancien éléments
334  fenêtre(index) = *iptr++;
335  index = (index + 1) % (K/R);
336 
337  // Pour chaque échantillon d'entrée, calcule R sorties différentes
338 
339  for(auto i = 0; i < R; i++)
340  {
341  T sum = 0;
342  auto cptr = coefs.data() + (R - 1) - i;
343  T *wptr = fenêtre.data() + index;
344 
345  int K1, K2;
346 
347  // Number of samples at end of delay line
348  K1 = K/R - index;
349  // Number at begin of delay line
350  K2 = K/R - K1;
351 
352  for(auto i = 0; i < K1; i++)
353  {
354  sum += *wptr++ * *cptr;
355  cptr += R;
356  }
357 
358  wptr = fenêtre.data(); // Restart at beginning of delay line
359  for(auto i = 0; i < K2; i++)
360  {
361  sum += *wptr++ * *cptr;
362  cptr += R;
363  }
364 
365  *optr++ = sum;
366  }
367  }
368  }
369 };
