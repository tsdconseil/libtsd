7 float goertzel(const ArrayXf &x, float f)
8 {
9  int n = x.rows();
10  float c = cos(2 * π * f);
11  float w0 = 0, w1 = 0;
12  float en = x.square().sum();
13 
14  // Dénominateur
15  for(auto i = 0; i < n; i++)
16  std::tie(w0, w1) = std::pair(2 * c * w0 - w1 + x(i), w0);
17 
18  // i = 0 : w0 = x0, w1 = 0
19  // i = 1 : w0 = ..., w1 = x0
20  // ....
21  // i = N-1 : ...
22 
23  // Dernière étape, avec xn = 0 :
24  // s_N = ...
25  std::tie(w0, w1) = std::pair(2 * c * w0 - w1, w0);
26 
27  // Normalisation par rapport à l'énergie du signal
28  return 2 * (w0 * w0 - 2 * c * w0 * w1 + w1 * w1) / (en * n);
29 }

50  {

55

59  cnt++;
60 
61  if(cnt >= R)
62  {

65

67  // Normalisation par rapport à l'énergie du signal
