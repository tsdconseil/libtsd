241 template<typename derived>
242 Vecteur<typename derived::Scalar> fftshift(const Eigen::ArrayBase<derived> &X)
243 {
244  auto n = X.rows();
245  Vecteur<typename derived::Scalar> res = Vecteur<typename derived::Scalar>::Zero(n);
246  if((n & 1) == 0)
247  {
248  res.tail(n/2) = X.head(n/2);
249  res.head(n/2) = X.tail(n/2);
250  }
251  else
252  {
253  res.tail(1+n/2) = X.head(1+n/2);
254  res.head(n/2) = X.tail(n/2);
255  }
256  return res;
257 }

758 template<typename derived>
759 std::tuple<ArrayXf, ArrayXf> psd(const Eigen::ArrayBase<derived> &x)
760 {
761  ArrayXf fen = tsd::filtrage::fenetre("hn", x.rows(), false);
762  auto xf = x * fen;
763 
764  if constexpr(est_complexe<typename derived::Scalar>())
765  {
766  ArrayXf Y = 10.0f * (fft(xf)).abs2().log10();
767  return {psd_freqs(x.rows(), true), fftshift(Y)};
768  }
769  else
770  {
771  ArrayXf Y = 10.0f * (rfft(xf)).abs2().log10();
772  return {psd_freqs(x.rows(), false), Y.head(Y.rows() / 2).eval()};
773  }
774 }
