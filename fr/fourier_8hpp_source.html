242 template<typename T>
243 Vecteur<T> fftshift(const Vecteur<T> &X)
244 {
245  auto n = X.rows();
246  Vecteur<T> res = Vecteur<T>::zeros(n);
247  if((n & 1) == 0)
248  {
249  res.tail(n/2) = X.head(n/2);
250  res.head(n/2) = X.tail(n/2);
251  }
252  else
253  {
254  res.tail(1+n/2) = X.head(1+n/2);
255  res.head(n/2) = X.tail(n/2);
256  }
257  return res;
258 }
