19 Vecteur<T> rifamp_naif(const Vecteur<T> &h, int L)
20 {
21  int N = h.rows();
22  int M = (N - 1) / 2;
23  tsd_assert_msg((N & 1) == 1, "rifamp_naif : nombre de coefficients impair attendu (N = {})", N);
24 
25  Vecteur<T> A(L);
26  for(auto i = 0; i < L; i++)
27  {
28  double omega = (i * π) / L;
29  A(i) = h(M);
30  for(auto k = 0; k < M; k++)
31  A(i) += 2 * h(k) * cos((M-k) * omega);
32  }
33  return A;
34 }

52  H = H.head(dim_fft/2).eval();

54  auto w = linspace(0,dim_fft/2-1,dim_fft/2) * π / (dim_fft / 2);
55 
56  // Corrige le facteur de normalisation de la fonction FFT
57  H *= sqrt((float)dim_fft);
58  H *= polar(M * w); // type I ou II
59

61  if(!symetrique)
62  H *= -complex<float>(0,1);
63

65 
66  auto fr = linspace(0, 0.5 - (0.5 / L), L);
67
