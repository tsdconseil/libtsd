12  std::tuple<MatrixXf, float> dare(const MatrixXf &A,
13  const MatrixXf &C, const MatrixXf &Q,
14  const MatrixXf &R,
15  float tolerance,
16  int iter_max)
17  {
18  MatrixXf P = Q,
19  At = A.transpose(),
20  Ct = C.transpose(),
21  Rinv = R.inverse();
22  float diff = 0;
23  for(auto i = 0; i < iter_max; ++i)
24  {
25  MatrixXf P2 = At * P * A - At * P * C * (R + Ct * P * C).inverse() * Ct * P * A + Q;
26  diff = ((P2 - P).array().abs().maxCoeff());
27  P = P2;
28  if(diff < tolerance)
29  break;
30  }
31  return {P, diff};
32  }

57 {
58  VectorXf y = f(x);
59  int ne = x.rows(), ns = y.rows();

63

340  {
341  // Proceed to one or more steps of the extended Kalman filter
342  //
343  // Calling Sequence
344  // [ekf,x] = ekf_process(ekf,y)
345  // [ekf,x] = ekf_process(ekf,y,dt)
346  //
347  // Parameters
348  // ekf: EKF filter object (see <link linkend="ekf_init">ekf_init</link>)
349  // y: Observations vector or matrix (one column for each time step)
350  // dt: optional time interval(s) (scalar or vector)
351  // x: Output state vector or matrix (one column for each time step)
352  //
353  // Description
354  // With the Extended Kalman Filter (EKF), the Kalman equations can be applied to a non-linear model (suboptimal solution).
355  // To do this, the system matrix <latex>$A$</latex> (matrix governing system evolution) and <latex>$C$</latex> (matrix governing system observations) are taken as the local derivative of the system and observation functions.
356  //
357  // If the user has not provided these Jacobians in <link linkend="ekf_init">ekf_init</link>, then numerical derivatives are computed (using the Scilab build-in function <link linkend="numderivative">numderivative</link>).
358  //
359  // See also
360  // ekf_init
361  // ekf_check
362  //
363  // Authors
364  // J.A., full documentation available on <ulink url="http://www.tsdconseil.fr/log/bayes">http://www.tsdconseil.fr/log/bayes</ulink>
365 
366  //if(ekf.ssm.subtype == 'linear')
367  // [ekf,s] = kalman_process(ekf,vobs);
368  // return;
369  //end;
370 
371 
372 
373  // Mise à jour intervalle de temps ?
374  // (pour échantillonnage non régulier)
375  //if(argn(2) > 2)
376  // ekf.ssm.dt = varargin(1);
377  //end;
378 
379  // Jacobienne de la fonction de transition (à partir de l'état précédent)
380  //MatrixXf Ak;
381  //if(!ssm.Jf)
382  // Ak = jacobienne_num(ssm.f, state);
383  //else
384  MatrixXf Ak = ssm->Jf(etat);
385 
386 
387  // Estimation a priori de l'état suivant
388  VectorXf xm = ssm->f(etat);
389 
390  if(mode_debug)
391  msg("etat predit : {}", xm);
392 
393  // Jacobienne de la fonction d'observation
394  MatrixXf Ck = ssm->Jg(xm);
395  //if(ssm.Jg)
396  // Ck = ssm.Jg(xm);
397  //else
398  // Ck = jacobienne_num(ssm.g, xm);
399 
400  if(mode_debug)
401  msg("Ck : {}", Ck);
402 
403  // Observation a priori (à partir du nouvel état !)
404  VectorXf vobs_th = ssm->g(xm);
405 
406  if(mode_debug)
407  {
408  msg("obs predit : {}", vobs_th);
409  msg("obs reelle : {}", vobs);
410  }
411 
412  // A priori estimation of the error covariance matrix
413  MatrixXf Pm = Ak * P * Ak.transpose() + ssm->Q;
414 
415  if(mode_debug)
416  msg("P predit : {}", Pm);
417 
418  // Optimal gain of the filter
419  if((Pm.isZero()) && (ssm->R.isZero()))
420  K = Ck.transpose().lu().inverse();
421  else
422  K = (Pm*Ck.transpose()) * (Ck * Pm * Ck.transpose() + ssm->R).lu().inverse();
423 
424  if(mode_debug)
425  msg("K : {}", K);
426 
427  // A posteriori estimation of the state variables
428  VectorXf x = xm + K * (y - vobs_th);
429 
430  if(mode_debug)
431  msg("etat post : {}", x);
432 
433  etat = x;
434 
435  // A posteriori estimation of the error covariance matrix:
436  P = (MatrixXf::Identity(ssm->ns,ssm->ns)-K*Ck)*Pm;
437 
438  if(mode_debug)
439  {
440  msg("K*Ck : {}", K*Ck);
441  msg("Gauche : {}", MatrixXf::Identity(ssm->ns,ssm->ns)-K*Ck);
442  msg("P post : {}", P);
443  }
444  k++;
445 
446  return x;
447  }

457  // Initialization of a standard Kalman filter object
458  FiltreKalman(sptr<SSMLineaire> &ssm, const Eigen::VectorXf &x0, const Eigen::MatrixXf &p0)
459  {
460  this->ssm = ssm;
461  if(x0.rows() == 0)
462  etat = Eigen::VectorXf::Zero(ssm->ns);
463  else
464  etat = x0;
465  // Matrice de covariance de l'état initial
466  if(p0.rows() == 0)
467  P = Eigen::MatrixXf::Identity(ssm->ns,ssm->ns);
468  else
469  P = p0;
470  tsd_assert_msg(etat.rows() == ssm->ns,
471  "Filtre de Kalman : nombre de lignes incorrect pour le vecteur état initial ({}, alors que ns={}).", etat.rows(), ssm->ns);
472  tsd_assert_msg((P.rows() == ssm->ns) && (P.cols() == ssm->ns),
473  "Filtre de Kalman : la matrice de covariance sur l'état initial devrait être de dimension {}x{}.", ssm->ns, ssm->ns);
474  }
475 
476 
477  // y: Current observations (column vector or matrix with one column for each time step)
478  // u: Current inputs (column vector or matrix with one column for each time step)
479  // x0: Initial state estimate (default is 0 or previous state if kalman_process already called)
480  // p0: Covariance matrix on the initial state (default is eye matrix, or previously estimated covariance matrix is kalman_process already called)
481  // retourne x : Current estimate of the state (column vector or matrix with one column for each time step)
482  VectorXf step(const VectorXf &y, const VectorXf &u)
483  {
484  auto sl = std::dynamic_pointer_cast<SSMLineaire>(ssm);
485 
486  // Estimation a priori de l'état suivant
487  VectorXf xm = sl->A * etat;
488 
489  if(sl->B.rows() > 0)
490  xm += sl->B * u;
491 
492  // Observation a priori
493  VectorXf vobs_th = sl->C * xm;
494 
495  if(sl->D.rows())
496  vobs_th += sl->D * u;
497 
498  // A priori estimation of the error covariance matrix
499  MatrixXf Pm = sl->A * P * sl->A.transpose() + sl->Q;
500 
501  MatrixXf K;
502 
503  // Optimal gain of the filter
504  if((Pm.isZero()) && (sl->R.isZero()))
505  K = sl->C.inverse();
506  else
507  K = (Pm * sl->C.transpose()) * (sl->C * Pm * sl->C.transpose() + sl->R).lu().inverse();
508 
509  // A posteriori estimation of the state variables
510  VectorXf x = xm + K * (y - vobs_th);
511 
512  etat = x;
513 
514  // A posteriori estimation of the error covariance matrix:
515  P = (Eigen::MatrixXf::Identity(sl->ns,sl->ns) - K * sl->C) * Pm;
516 
517  return etat.array();
518  }
519 };
