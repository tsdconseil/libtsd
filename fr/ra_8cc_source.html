40  {

42 
43  if(n == 0)
44  {
45  y.resize(0);
46  return;
47  }
48 
49  int tmp_len = ceil(ratio * n) + 10;
50 
51  Vecteur<T> temp(tmp_len);
52 
53  auto j = 0;
54  auto optr = temp.data();
55  auto iptr = x.data();
56 
57  for(auto i = 0; i < n; i++)
58  {
59  // Rotation de la fenêtre
60  // TODO : ça serait pas mal que les interpolateurs puissent fonctionner de manière circulaire
61  // pour éviter cette rotation
62  fenetre.head(nfen-1) = fenetre.tail(nfen-1).eval();
63  fenetre(nfen-1) = *iptr++;

67  // phase = index entre deux échantillons

71  j++;

73  }
74  phase--;
75  } // end for i
