10 // IMU (Inertial Measuring Unit) generic model
11 //
12 // Calling Sequence
13 // ssm = model_imu();
14 // ssm = model_imu(cfg);
15 //
16 // Parameters
17 // cfg: IMU model configuration (see <link linkend="model_imu_cfg">model_imu_cfg()</link>)
18 // ssm: State / space model
19 //
20 // Description
21 // Generic IMU model with Euler angles parametrization, and with the following
22 // (optionnal) sensors:
23 // <itemizedlist>
24 // <listitem><para>Accelerometer (3 axis)</para></listitem>
25 // <listitem><para>Gyroscope (3 axis)</para></listitem>
26 // <listitem><para>Magnetometer / compas (3 axis)</para></listitem>
27 // <listitem><para>Barometer</para></listitem>
28 // <listitem><para>GPS</para></listitem>
29 // </itemizedlist>
30 //
31 // This model can be used either for simulation (see <link linkend="ssm_simu">ssm_simu</link>)
32 // or for orientation / position estimation from experimental data (for instance using an EKF filter, see <link linkend="ekf_init">ekf_init</link>).
33 //
34 // The model can use as a state vector either only the object orientation, or both the
35 // object orientation and localization.
36 // So it is two parts, where each part is optionnal (according to the configuration):
37 // <itemizedlist>
38 // <listitem><para><emphasis role="bold">Position: </emphasis> <latex>$(x,y,z,\dot{x},\dot{y},\dot{z},\ddot{x},\ddot{y},\ddot{z})$</latex></para></listitem>
39 // <listitem><para><emphasis role="bold">Orientation: </emphasis> <latex>$(\phi,\theta,\psi,\dot{\phi},\dot{\theta},\dot{\psi})$</latex></para></listitem>
40 // </itemizedlist>
41 //
42 // The Observation vector is in this order: <latex>$(a_x,a_y,a_z,\omega_x,\omega_y,\omega_y,m_x,m_y,m_z,p_{baro},x_{gps},y_{gps},z_{gps})$</latex>,
43 // where each element is optionnal, according to the configuration.
44 //

258  VectorXf f(const VectorXf &x) const
259  {
260  VectorXf x2(ns);
261 
262  if(config.estim_pos)
263  {
264  x2.head(3) = x.head(3) + δt * x.segment(3, 3);
265  x2.segment(3,3) = x.segment(3,3) + δt * x.segment(6,3);
266  x2.segment(6,3).setZero();//x(7:9); // Accelerations nulles (bruit sans mémoire)
267  }
268 
269  if(config.estim_orient)
270  {
271  int io = iorient;
272  x2.segment(io,3) = x.segment(io,3) + δt * x.segment(io+3,3);
273  x2.segment(io+3,3) = x.segment(io+3,3); // Vitesses angulaires constantes
274  }
275  return x2;
276  }

278  VectorXf g(const VectorXf &x) const
279  {
280  int io = iorient;
281  float cp = cos(x(io)), sp = sin(x(io)), ct = cos(x(io+1)), st = sin(x(io+1));
282 
283  // Matrice de rotation repère absolu -> repère inertiel
284  auto Rmat = (Cardan{x(io),x(io+1),x(io+2)}).mat_rotation();
285 
286  VectorXf y(no);
287 
288  int id = 0;
289  if(config.acc_available)
290  {
291  Eigen::Vector3f v = Eigen::Vector3f::Zero();
292  v(2) = G;
293 
294  if(config.estim_pos)
295  v += x.segment(6, 3);
296 
297  y.head(3) = Rmat * v;
298  id += 3;
299  }
300  if(config.gyro_available)
301  {
302  float dϕ = x(io+3), dθ = x(io+4), dψ = x(io+5);
303  y(id) = dϕ - st * dψ;
304  y(id+1) = cp * dθ + sp * ct * dψ;
305  y(id+2) = cp * ct * dψ - sp * dθ;
306  id += 3;
307  }
308  if(config.mag_available)
309  {
310  Eigen::Vector3f v;
311  v << 0, 1, 0;
312  y.segment(id,3) = Rmat * v; // En supposant force normalisée à 1 gauss.
313  id += 3;
314  }
315  if(config.baro_available)
316  {
317  y(id) = x(2);
318  id++;
319  }
320  if(config.gps_available)
321  {
322  y.segment(id,3) = x.head(3);
323  id++;
324  }
325  return y;
326  }
