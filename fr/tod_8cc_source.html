111  {

114 
115  Vecteur<T> tmp(half);
116 
117  // Stockage des coefs pairs
118  for(auto i = 0; i < half; i++)

120 
121  // Ecriture des coefs impairs
122  // x[n-1] := x[n-1] (to remove)
123  // x[n-2] := x[n-3]
124  // ....
125  // x[n-half] := x[n-(n+1)+1] = x[0] si n impair
126  // x[n-half] := x[n-n+1] = x[1] si n pair

130  for(auto i = 1; i <= max; i++)
131  x(n-i) = x(n-(2*i)+1-dec);
132 
133  x.head(half) = tmp;
