200 static std::vector<std::string> split(const std::string& str, const std::string& delim)
201 {
202  std::vector<std::string> tokens;
203  size_t prev = 0, pos = 0;
204  do
205  {
206  pos = str.find(delim, prev);
207  if(pos == std::string::npos)
208  pos = str.length();
209  std::string token = str.substr(prev, pos-prev);
210  if(!token.empty())
211  tokens.push_back(token);
212  prev = pos + delim.length();
213  }
214  while (pos < str.length() && prev < str.length());
215  return tokens;
216 }

286 double grégorien_vers_jours(int année, double jour_année)
287 {
288  int64_t a1 = année - 1;
289 
290  return
291  // Nombre de jours de l'année précédente :
292  365 * a1
293  // Nombre de jours supplémentaires dues aux années bissextiles :
294  + a1 / 4
295  // Mais les années multiples de 100 ne sont pas bissextiles :
296  - a1 / 100

357 double DateHeure::temps_sidéral_Greenwich() const
358 {
359  // https://lweb.cfa.harvard.edu/~jzhao/times.html
360 
361  // Nb jours Juliens du minuit précédent
362  auto jd0 = floor(nb_jours_Julien() + 0.5) - 0.5;
363  auto t = (jd0 - 2451545.0) / 36525.0;
364  auto jdf = nb_jours_Julien() - jd0;
365  auto gt = 24110.54841 + t * (8640184.812866 + t * (0.093104 - t * 6.2E-6));
366  gt += jdf * 1.00273790935 * 86400.0;
367  return wrap_2pi(deg2rad(gt * 360.0 / nbsecs_par_jour));
368 }

467 Calendrier DateHeure::calendrier() const
468 {
469  int nb_jours = ntics / TICS_JOUR;
470 
471  // Décompte les cycles de 400 ans
472  int ncycles_400_ans = nb_jours / nb_jours_cycle_400_ans;
473  nb_jours %= nb_jours_cycle_400_ans;
474 
475  // Décompte les cycles de 100 ans
476  // 1,2,3,...99, 100,101,...,199, ....
477  int ncycles_100_ans = nb_jours / nb_jours_cycle_100_ans;
478  if (ncycles_100_ans == 4)
479  // 31 décembre de la dernière année bissextile
480  ncycles_100_ans = 3;
481  nb_jours -= ncycles_100_ans * nb_jours_cycle_100_ans;
482 
483  // Décompte les cycles de 4 ans
484  int ncycles_4_ans = nb_jours / nb_jours_cycle_4_ans;
485  nb_jours %= nb_jours_cycle_4_ans;
486 
487  // Décompte les années qui restent (dans le cycle de 4 ans en cours)
488  // Un cycle de 4 ans =
489  // 1, 2, 3, 4
490  // ...
491  // 2021, 2022, 2023, 2024
492  // Soit en général :
493  // [365, 365, 365, 366]
494  // 0 <= nb jours < 3 * 365 + 366
495  int nb_ans = 0;
496  if(nb_jours < 365)
497  ;
498  else if(nb_jours < 2 * 365)
499  {
500  nb_ans = 1;
501  nb_jours -= 365;
502  }
503  else if(nb_jours < 3 * 365)
504  {
505  nb_ans = 2;
506  nb_jours -= 2 * 365;
507  }
508  else
509  {
510  nb_ans = 3;
511  nb_jours -= 3 * 365;
512  }
513 
514  int année = (ncycles_400_ans * 400) + (ncycles_100_ans * 100)
515  + (ncycles_4_ans * 4) + nb_ans + 1;
516 
517  // Nombre de jours pour chaque mois
518  auto ptr = est_bissextile(année) ? nb_jours_par_mois[1] : nb_jours_par_mois[0];
519 
520  int mois = 0;
521  while((nb_jours >= ptr[mois]) && (mois <= 11))
522  nb_jours -= ptr[mois++];
523 
524  return {année, mois + 1, nb_jours + 1};
525 }
