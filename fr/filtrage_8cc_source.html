812  {

814  pool.erase(pool.begin());
815

817  // z² -(z(i)+z(i+1))*z + z(i)*z(i+1)

837  continue;

840  auto sp0 = -(p(j) + p(k)), pp0 = p(j)*p(k);

846  berr = err;
847  sz = sz0;
848  pz = pz0;
849  sp = sp0;
850  pp = pp0;
851  }
852  }

898  }
