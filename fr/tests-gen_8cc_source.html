124 int fait_tests(int argc, const char *argv[], std::vector<Test> &tests)
125 {
126  // Pour éviter la fenêtre Windows en cas d'échec d'assertion
127  // (celle-ci empêche le point d'arrêt gdb)
128  _set_error_mode(_OUT_TO_STDERR);
129 
130  int opt;
131  msg("Tests automatiques libtsd...");
132 
133  tsd::vue::stdo.def_dossier_sortie("./build/test-log");
134 
135  while ((opt = getopt(argc, (char * const *) argv, "t:hald")) != -1)
136  {
137  std::string nom;
138 
139  if(optarg != nullptr)
140  nom = optarg;
141 
142  switch (opt)
143  {
144  case 'd':
145  tests_debug_actif = true;
146  break;
147  case 't':
148  {
149  for(auto i = 0u; i < tests.size(); i++)
150  {
151  if(tests[i].nom == nom)
152  {
153  int res = teste(i, tests);
154  fmt::print("Fin des tests.\n");
155  return res;
156  }
157  }
158  msg_erreur("Test non trouvé : [{}]", nom);
159  return -1;
160  }
161  case 'h':
162  case 'a':
163  case 'l':
164  {
165  fmt::print("Liste des tests:\n");
166  for(auto &t: tests)
167  fmt::print(" - {}\n", t.nom);
168  return 0;
169  }
170  }
171  }
172 
173  for(auto i = 0u; i < tests.size(); i++)
174  {
175  int res = teste(i, tests);
176  if(res)
177  return res;
178  }
179 
180 
181  fmt::print("Fin des tests.\n");
182 
183  tsd::vue::stdo.fin();
184 
185  return 0;
186 }
