32 static float rcs(float t, float β)
33 {
34  auto δ = abs(t) - 1.0f / (4*β);
35  if(t == 0)
36  return (1-β) + 4*β/π;
37  else if ((β > 0) && (abs(δ) < 1e-7))
38  return (β/sqrt(2)) * ((1+(2/π))*sin(π/(4*β))+(1-(2/π))*cos(π/(4*β)));
39  return (sin(π*t*(1-β)) + 4 * β * t * cos(π * t * (1+β)))
40  / (π * t * (1 - carré(4 * β * t)));
41 }

165 ArrayXf design_rif_cs(int n, float β, float fcoupure)
166 {
167  int m = n / 2;
168  ArrayXf h(n);
169  for(auto i = 0; i < n; i++)
170  {
171  // ncoefs = 7 --> m = 3
172 
173  // Pour t entre -m et m :
174  // - 0 -> -m
175  // - n-1 -> m
176  // n = 3 -> -1, 0, 1
177  float t = i - m;
178  // n = 4 -> -2, 1, 0, 1
179  // La fonction cs s'annule pour t entier non nul, si fcut = 0.5 => Dirac (filtre neutre)
180  h(i) = cs(t * fcoupure * 2, β);
181  }
182 
183  return h / h.sum();
184 }
