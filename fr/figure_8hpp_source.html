213  // Dessine dans un sous-rectangle du Canva passé en paramètre
214  // rdi est spécifiée en coordonnées virtuelle de l'ancien canva.
215  // Exemples pour une grille :
216  // Canva c;
217  // c.clip({0,0,0.5,0.5});
218  // figure1.rendre(c);
219  // -> Dans figure::rendre
220  // auto c1 = c.vue({xmin,ymin,xmax-xmin,ymax-ymin})
221  // c1.ligne(x(i), y(i), x(i+1), y(i+1))
222 
223  // clip permet de modifier la cible des dessins
224  // vue permet de modifier la source des dessins (système de coordonnées)
225 
226  // x = x_clip + rdi_clip.l * (xv - rdi_view.x) / rdi_view.l
227 
228  // Soit :
229  // x = alpha + beta * xv
230 
231 
232  // Donc dans l'ordre :
233  // - canva initial, de dimension e.g. 640 x 480
234  // - vue initiale = aussi 640 x 480
235  // - C'est à dire qu'initialement, x virtuel = x concret
236  // - On pose alpha = 1, beta = 0 (coefs de conversion virtuel -> concret)
237  // - Effet de clip :
238  // beta_x += rdi.x, alpha_x = inchangé
239  // - Effet de view :
240  // beta_x += -rdi.x / rdi.l, alpha_x *= 1 / rdi.l
241
