233 template<typename derived>
234 Vecteur<typename derived::Scalar> fftshift(const Eigen::ArrayBase<derived> &X)
235 {
236  auto n = X.rows();
237  Vecteur<typename derived::Scalar> res = Vecteur<typename derived::Scalar>::Zero(n);
238  if((n & 1) == 0)
239  {
240  res.tail(n/2) = X.head(n/2);
241  res.head(n/2) = X.tail(n/2);
242  }
243  else
244  {
245  res.tail(1+n/2) = X.head(1+n/2);
246  res.head(n/2) = X.tail(n/2);
247  }
248  return res;
249 }

720 std::tuple<ArrayXf, ArrayXf> psd(const Eigen::ArrayBase<derived> &x)
721 {
722  ArrayXf fen = tsd::filtrage::fenetre("hn", x.rows(), false);
723  auto xf = x * fen;
724 
725  if constexpr(est_complexe<typename derived::Scalar>())
726  {
727  ArrayXf Y = 10.0f * (fft(xf)).abs2().log10();
728  return {psd_freqs(x.rows(), true), fftshift(Y)};
729  }
730  else
731  {
732  ArrayXf Y = 10.0f * (rfft(xf)).abs2().log10();
733  return {psd_freqs(x.rows(), false), Y.head(Y.rows() / 2).eval()};
734  }
