17  std::vector<int> parse_liste_entiers(const std::string &str)
18  {
19  std::vector<int> res;
20  auto s = str.c_str();
21  res.clear();
22  int current = 0;
23  if(!est_deci(s[0]))
24  return std::vector<int>();
25  while(strlen(s) > 0)
26  {
27  if(est_deci(s[0]))
28  {
29  current = current * 10 + (s[0] - '0');
30  }
31  else
32  {
33  auto c = s[0];
34  if(((c == '.') || (c == ',') || (c == '/') || ((c == ':'))) && (strlen(s) > 1))
35  {
36  res.push_back(current);
37  current = 0;
38  }
39  else
40  {
41  return std::vector<int>();
42  }
43  }
44  s++;
45  }
46  res.push_back(current);
47  return res;
48  }

157

159  int nl = lignes.size();
160

162

176  }
177 
178  if((dim_max.l < 1) || (dim_max.h < 1))
179  {

181  return O;

183 
184

186  Dim dim_totale;

189  {
190  nitr++;

192  for(auto i = 0u; i < lignes.size(); i++)
193  {
194  ypos[i] = dim_totale.h;
195  Dim dim = O.texte_dim(lignes[i], scale);
196 
197  dim.h += 5; // Ajoute un peu d'espace entre les lignes
198 
199  dim_totale.l = std::max(dim.l, dim_totale.l);
200 
201  ydim[i] = dim.h;
202  xdim[i] = dim.l;
203  dim_totale.h += dim.h;
204  }
205  if(config.dim_max.l == -1)
206  break;
207  if((dim_totale.l <= dim_max.l) && (dim_totale.h <= dim_max.h))
208  break;
209  if(scale < 1e-3)
210  {
211  msg_avert("Image::affiche_texte({} lignes, [{}]). Echec placement @scale = {}. Dim max = {}.", lignes.size(), lignes[0], scale, dim_max);
212  return O;
213  }
214  scale *= 0.9;
215  }
