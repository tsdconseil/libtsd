226 {
227  int i, n = x.size() - 1, n2 = x2.rows();
228  Vecteur<T> B(n), C(n+1), D(n), Y(n), L(n+1), U(n+1), Z(n+1), R(n2);
229  ArrayXf H(n);
230  auto &A = y;
231 
232  H = diff(x);
233 
234  /*for(i = 0; i < n; i++)
235  {
236  H(i) = x(i+1) - x(i);
237  tsd_assert_msg(H(i) > 0, "interp_cspline: vecteur x croissant attendu");
238  }*/
239 
240  //for(i = 0; i <= n; i++)
241  //A(i) = vy(i);
242  //A = y;
243  //auto dy = diff(y);
244  //Y = (3 / H) * dy - (3 / H) * dy;
245 
246  for(i = 1; i < n; i++)
247  Y(i) = ((T) (3.0 / H(i))) * (y(i+1) - y(i)) - ((T) (3.0 / H(i-1))) * (y(i) - y(i-1));
248 
249  L(0) = U(0) = Z(0) = 0;
250 
251  for(i = 1; i < n; i++)
252  {
253  L(i) = ((T) 2) * (x(i+1) - x(i-1)) - H(i-1) * U(i-1);
254  U(i) = H(i) / L(i);
255  Z(i) = (Y(i) - H(i-1) * Z(i-1)) / L(i);
256  }
257  L(n) = Z(n) = C(n) = 0;
258 
259  for(i = n-1; i >= 0; i--)
260  {
261  C(i) = Z(i) - U(i) * C(i+1);
262  B(i) = (y(i+1) - y(i)) / H(i) - H(i) * (C(i+1) + ((T) 2) * C(i)) / ((T) 3.0);
263  D(i) = (C(i+1) - C(i)) / (3 * H(i));
264  }
265  // Resample
266  int id = 0;
267  for(i = 0; i < x2.rows(); i++)
268  {
269  if(x2(i) <= x(0))
270  R(i) = y(0);
271  else if(x2(i) >= x(n))
272  R(i) = y(n);
273  else
274  {
275  for(; x(id) < x2(i); id++)
276  ;
277  // x2(i) compris dans [x(id-1) ; x(id)]
278  float t = x2(i) - x(id-1);
279  R(i) = A(id-1) + B(id-1) * t + C(id-1) * t * t + D(id-1) * t * t * t;
280  }
281  }
282  return R;
283 }

321 void calc_cubic_spline(float h[4], float t)
322 {
323  h[0] = (1 + 2 * t) * (t - 1)*(t - 1);
