285  {
286  Poly<T> res;
287  res.vname = vname;
288  res.mode_racines = mode_racines;
289 
290  auto s1 = *this, s2 = s;
291 
292  if(s1.mode_racines && !s2.mode_racines)
293  {
294  s1 = s1.vers_coefs();
295  }
296  else if(!s1.mode_racines && s2.mode_racines)
297  {
298  s2 = s2.vers_coefs();

300 
301  tsd_assert(s1.mode_racines == s2.mode_racines);
302

306  res.mlt = s1.mlt * s2.mlt;

311  {
312  Poly<T> prod = s1;

315  for(auto i = 0; i < s1.coefs.rows(); i++)
316  prod.coefs(i+k) = s1.coefs(i) * s2.coefs(k);
317  res = res + prod;
318  }
