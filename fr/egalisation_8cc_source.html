75  // Creation of an equalizer object (LMS, CMA, ...)
76  //
77  // Calling Sequence
78  // eq = equalizer_init(wf); // Default equalizer (DDE, K = 1)
79  // eq = equalizer_init(wf,K,errf,struct,gain,N1[,N2]]);
80  //
81  // Parameters
82  // wf: Waveform specification (see <link linkend="wf_init">wf_init</link>)
83  // K: Input over-sampling ratio (number of samples per symbol)
84  // errf: Error function (either 'slicer' or 'cma')
85  // struct: Equalizer structure (either 'dde' or 'dfe')
86  // gain: Update gain (determine the convergence speed)
87  // N1: Number of taps of the direct FIR equalization filter
88  // N2: Number of taps of the feedback FIR equalization filter (only for DFE equalization)
89  // eq: Resulting equalizer object (can be used with <link linkend="equalizer_process">equalizer_process</link>)
90  //
91  //
92  // Description
93  // Initialize an equalizer object, ever symbol sampled (K = 1), or
94  // fractionnaly sampled (K > 1). Supported equalization structures are the
95  // following:
96  // <itemizedlist>
97  // <listitem><para><emphasis role="bold">Decision Directed Equalization (DDE)</emphasis> <varname>(strucuture = 'dde'):</varname> A direct FIR filter is tuned (every symbol period) to minimize the output error.</para></listitem>
98  // <listitem><para><emphasis role="bold">Decision Feedback Equalization (DFE)</emphasis> <varname>(strucuture = 'dfe'):</varname> Both a direct FIR filter (working at <latex>$K\cdot f_{symb}$</latex>) and a feedback filter (working at symbol frequency) on the decision (slicer) outputs are used.</para></listitem>
99  // </itemizedlist>
100  //
101  // Possible cost function are the following:
102  // <itemizedlist>
103  // <listitem><para><emphasis role="bold">Slicer</emphasis> <varname>(errf = 'slicer'):</varname> <latex>$E=(d-y)^2$</latex>. Note: with this error function, the algorithm is better known as <emphasis role="bold">LMS (Least Mean Square).</emphasis></para></listitem>.
104  // <listitem><para><emphasis role="bold">Constant modulus algorithm (CMA)</emphasis> <varname>(errf = 'cma'):</varname> <latex>$E=\left(R-|y|^2\right)^2$</latex></para></listitem>
105  // </itemizedlist>
106  // For a complete example, see the <link linkend="equalizer_process">equalizer_process</link> function.
107  //
108  // See also
109  // equalizer_process
110  // wf_init
111  //
112  // Authors
113  // J. Arzi, full documentation available on <ulink url="http://www.tsdconseil.fr/log/sct">http://www.tsdconseil.fr/log/sct</ulink>
114 
115  this->wf = wf;
116 
117 
118  if((s_errf != "cma") && (s_errf != "slicer"))
119  {
120  msg_erreur("init égaliseur: supported error function = \"slicer\" or \"cma\" (requested : \"{}\").", s_errf);
121  return;
122  }
123 
124  if((s_structure != "dde") && (s_structure != "dfe"))
125  {
126  msg_erreur("init égaliseur: supported structure = \"dde\" or \"dfe\" (requested : \"{}\").", s_structure);
127  return;

131  errf = s_errf == "slicer" ? SLICER : CMA;
132 
133 
134  if(structure == DFE)

137  h_arriere = ArrayXf::Zero(N2);
138  wnd = ArrayXf::Zero(N1);
139  wnd_deci = ArrayXf::Zero(N2);
140  cnt = 0;
141  }
142  else if(structure == DDE)

144  h = vconcat(ArrayXf::Zero(N1-1), ArrayXf::Ones(1));
145  wnd = ArrayXf::Zero(N1);
146  }
147  else

149  msg_erreur("equalizer_init: supported structures = 'dde' or 'dfe'.");

151  }
152

175  void step_interne(IArrayXcf x, ArrayXcf &y, ArrayXf &err)
176  {
177  // Equalization of an input signal
178  //
179  // Calling Sequence
180  // [eq,y] = equalizer_process(eq,x);
181  // [eq,y,err] = equalizer_process(eq,x);
182  //
183  // Parameters
184  // x: input signal to be equalized
185  // y: output signal
186  // err: Error for each symbol
187  //
188  // Description
189  // Perform CMA or LMS equalization, using DDE (Decision directed) or DFE (Decision feedback) structure.
190  //
191  // <refsection><title>Example</title></refsection>
192  // <programlisting>
193  // fs = 4; fi = 0; fsymb = 1;
194  // wf = wf_init('qpsk');
195  // eq = equalizer_init(wf, fs/fsymb, 'cma', 'dfe', 0.01,21);
196  // h = [1 0.1 -.1 0.15 0.05];
197  // mod = mod_init(wf, fs,fi,fsymb);
198  // [mod,x] = mod_process(mod,prbs(2000));
199  // y = convol(h,x);
200  // y = awgn(y, 0.02, 'c');
201  // [eq,z] = equalizer_process(eq, y);
202  // </programlisting>
203  // <mediaobject><imageobject><imagedata fileref="ex-dfe-cma-const.png" format="PNG"/></imageobject><caption><para>Example: DFE / CMA equalization</para></caption></mediaobject>
204  //
205  // See also
206  // equalizer_init
207  //
208  // Authors
209  // J. Arzi, full documentation available on <ulink url="http://www.tsdconseil.fr/log/sct">http://www.tsdconseil.fr/log/sct</ulink>
210 
211  auto n = x.rows();
212  y = ArrayXf::Zero(n);
213  err = ArrayXf::Zero(n);
214 
215  if(structure == DDE)
216  {
217  for(auto i = 0; i < n; i++)
218  {
219  // wnd = [wnd(2:N1) ; x(i)];
220  wnd.head(N1-1) = wnd.tail(N1-1).eval();
221  wnd(N1-1) = x(i);
222 
223  auto egalise = (h * wnd).sum();
224  auto d = wf->lis_symbole(wf->symbole_plus_proche(egalise));
225  float e;
226 
227  if(errf == CMA)
228  {
229  e = g2 - std::abs(egalise) * std::abs(egalise);
230  h = h + gain * e * (wnd * std::conj(egalise)).real();
231  }
232  else
233  {
234  auto ec = d - egalise;
235  e = std::abs(ec);
236  h = h + gain * (std::conj(ec) * wnd).real();
237  }
238  y(i) = egalise;
239  err(i) = e;
240  }
241  }
242  else if(structure == DFE)
243  {
244  auto j = 1;
245  for(auto i = 0; i < n; i++)
246  {
247  wnd.head(N1-1) = wnd.tail(N1-1).eval();
248  wnd(N1-1) = x(i);
249  cnt++;
250  if(cnt != K)
251  continue;
252 
253  cnt = 0;
254  auto retroaction = (h_arriere * wnd_deci).sum();
255  auto egalise = (h_avant * wnd).sum() + retroaction;
256 
257  auto d = wf->lis_symbole(wf->symbole_plus_proche(egalise));
258  float e;
259  if(errf == CMA)
260  {
261  e = g2 - std::abs(egalise * egalise);
262  h_avant += gain * e * (wnd * std::conj(egalise)).real();
263  h_arriere += gain * e * (wnd_deci * std::conj(egalise)).real();
264  }
265  else
266  {
267  auto ec = d - egalise;
268  e = std::abs(ec);
269  h_avant += gain * (std::conj(ec) * wnd).real();
270  h_arriere += gain * (std::conj(ec) * wnd_deci).real();
271  }
272 
273  err(j) = e;
274  y(j++) = egalise;
275 
276  wnd_deci.head(N2-1) = wnd_deci.tail(N2-1).eval();
277  wnd_deci(N2-1) = d;
278 
279  } // for i
280 
281  y.resize(j);
282  err.resize(j);
283  } // structure DFE
284  }

296 ArrayXf egaliseur_zfe(IArrayXf h, int n)
297 {
298  int m = h.rows();
299 
300  if(n < m)
301  {
302  msg_erreur("egaliseur_zfe(h[{}],n={}): n doit être supérieur ou égal à la longueur de la réponse du canal.", m, n);
303  return ArrayXf();
304  }
305 
306  // (1) Calcule un délais
307  // Suppose que h et g sont centrées, delai = m/2 + n/2
308  auto d = (int) round(m/2 + n/2);
309 
310  // (2) Construit la matrice de filtrage : n+m-1 lignes, n colonnes
311  Eigen::MatrixXf F = Eigen::MatrixXf::Zero(n+m-1,n);
312 
313  // Ligne 0 à m-1
314  for(auto i = 0; i < m; i++)
315  F.block(i,0,1,i+1) = h.segment(0, i+1).reverse().transpose();//h(i+1:-1:1)';
316 
317  // Ligne m à n-1
318  for(auto i = m; i <= n-1; i++)
319  {
320  // i = m : col = [1...1+m-1]
321  // i = n-1: col = [n-m...n-1]
322  F.block(i,i+1-m,1,m) = h.reverse().transpose();//h($:-1:1)';
323  }
324 
325  // Ligne n à n+m-2
326  for(auto i = n; i < n+m-1; i++)
327  for(auto j = 1; j <= m-1-(i-n); j++)
328  F(i,n-j) = h(j+(i-n));
329 
330  Eigen::VectorXf δ = Eigen::VectorXf::Zero(n+m-1);
331  δ(d, 0) = 1.0f;
332 
333  //g = F \ δ;
334  ArrayXf g = F.colPivHouseholderQr().solve(δ).array();
335 
336  return g;
337 }
