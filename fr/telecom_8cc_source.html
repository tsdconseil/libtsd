48  CmpBitsRes cmp_bits(const BitStream &bs1, const BitStream &bs2)
49  {
50  CmpBitsRes res;
51 
52  ArrayXf b1 = bs1.vers_array(), b2 = bs2.vers_array();
53 
54  ArrayXcf b1c = b1, b2c = b2;
55  auto [b1a, b2a, dt, score] = tsd::fourier::aligne_entier(b1c, b2c);
56 
57  ArrayXf b1ar = b1a.real(), b2ar = b2a.real();
58 
59  ArrayXf b1ap = b1ar, b2ap = b2ar;
60 
61 
62  // TODO : utiliser plutot segment dans le comptage d'erreurs
63  if(b1a.rows() > 100)
64  {
65  auto n = b1a.rows();
66  b1ap = b1ar.tail(n - 50).head(n-100).eval();
67  b2ap = b2ar.tail(n - 50).head(n-100).eval();
68  }
69 
70  res.nerr = 0;
71  for(auto i = 0; i < b1ap.rows(); i++)
72  {
73  if(std::round(b1ap(i)) != std::round(b2ap(i)))
74  res.nerr++;
75  }
76 
77  //res.nerr = (b1ap-b2ap).abs().sum();
78 
79  if(b2a.rows() == 0)
80  res.ber = NAN;
81  else
82  res.ber = ((float) res.nerr) / b2ap.rows();
83 
84  msg("Cmp bits : dec = {}, nerr = {}, rows = {}, ber = {:.1e}", dt, res.nerr, b2ap.rows(), res.ber);
85 
86  res.decalage = dt;
87  res.score = score;
88  res.b0 = b1ar;
89  res.b1 = b2ar;
90 
91  if(0)
92  {
93  Figures f;
94  f.subplot().plot(b1, "hb", "Sequence d'entree ({} bits)", b1.rows());
95  f.subplot().plot(res.b0, "hb", "Sequence d'entree recalée");
96  f.subplot().plot(b2, "hb", "Sequence de sortie ({} bits)", b2.rows());
97  f.subplot().plot(res.b1, "hb", "Sequence de sortie recalée");
98  f.subplot().plot(res.b1 - res.b0, "hr", "Erreurs ({}, ber = {:.1e})", res.nerr, res.ber);
99  f.afficher(fmt::format("Comparaison bits (dec={}, s={})", dt,score));
100  }
