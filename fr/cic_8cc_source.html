27  FiltreCIC(const CICConfig &config, char mode = 'd')
28  {
29  this->config = config;
30  this->mode = mode;
31 
32  if((mode != 'd') && (mode != 'u') && (mode != 'i'))
33  echec("cic_init: le mode doit être 'd' ou 'i'.");
34  if(config.M != 1)
35  echec("cic_init: seulement M = 1 est supporté pour l'instant.");
36  mem_diff.setZero(config.N);
37  if(mode == 'd')
38  gain = 1 / pow(config.R*config.M,config.N);
39  else
40  gain = config.R / pow(config.R*config.M,config.N);
41  }

43  void step(const Eigen::Ref<const Vecteur<T>> x, Vecteur<T> &y)
44  {
45  Vecteur<Ti> xi = x.template cast<Ti>();
46 
50  if(mode == 'd')
51  {
52  // (1) Integration
53  for(auto i = 0; i < config.N; i++)
54  xi = cumsum(xi);
55 
56  // (2) Decimation
57  Vecteur<Ti> xd = sousech(xi, config.R);
58 
59  // (3) Peignes
60  for(auto i = 0; i < config.N; i++)
61  {
62  Vecteur<Ti> xp(xd.rows() + 1);
63  xp(0) = mem_diff(i);
64  xp.segment(1, xd.rows()) = xd;
65  mem_diff(i) = xd(xd.rows()-1);
66  xd = diff(xp);
67  }
68  y = xd.template cast<float>() * gain;
69  return;
70  }
74  // (1) Peignes
75  for(auto i = 0; i < config.N; i++)
76  {
77  Vecteur<Ti> xp(xi.rows() + 1);
78  xp(0) = mem_diff(i);
79  xp.segment(1,xi.rows()) = xi;
80  mem_diff(i) = x(xi.rows()-1);
81  xi = diff(xp);
82  }
83 
84  // (2) Upsampling
85  Vecteur<Ti> tmp = Vecteur<Ti>::Zero(xi.rows()*config.R);
86  auto map = Eigen::Map<Vecteur<Ti>,0,Eigen::InnerStride<>>(tmp.data(), tmp.rows() / config.R, Eigen::InnerStride<>(config.R));
87  map = xi;
88  xi = tmp;
89 
90  // (3) Integration
91  for(auto i = 0; i < config.N; i++)
92  xi = cumsum(xi);
93 
94  y = xi.template cast<float>() * gain;
95  }
96  };

105 FRat<float> design_cic(const CICConfig &config)
106 {
107  tsd_assert((config.R > 0) && (config.M > 0) && (config.N >= 0));
108 
109  FRat<float> h;
110  h.numer.coefs.setOnes(config.R * config.M);
111  h.numer.coefs /= (config.R * config.M);
112  return h.horner(FRat<float>::z().inv()).pow(config.N);
113 }

221 ArrayXf cic_freq(const CICConfig &config, const ArrayXf &f)
222 {
223  ArrayXf mag = ArrayXf::Ones(f.rows());
224  for(auto i = 0; i < f.rows(); i++)
225  {
226  float d = config.R*config.M*sin(π*f(i));
227  if(d != 0)
228  mag(i) = pow((float) abs(sin(config.R*config.M*π*f(i)) / d), (float) config.N);
229  }
230  return mag;

242  msg("R = {}, Fout = {:.2f} Hz, Fin = {:.2f} Hz.", config.R, Fout, Fin);

246  auto idx = trouve(analyse.fr > Fout / 2)[0];
247  msg("index = {} / {}.", idx, length(analyse.fr));

250 
251  msg("idx = {}", idx);

253  //scf(2); clf();
254 
255  // Ideal response of the FIR compensation filter = 1 / CIC response
256  ArrayXf idealc = 1.0f / fm2;
257 
258  // And, also, cut @ Fint Hz
259 
260  auto id = 1 + trouve(analyse.fr > Fcut)[0];
261 
262  msg("Fcut = {} Hz, id = {}", Fcut, id);
263 
264  tsd_assert((id > 1) && (id + 1 < analyse.fr.rows()));
265
