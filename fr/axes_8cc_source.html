212 // Calcul une valeur de tic vérifiant tic >= tic_min,
213 // et "simple" en terme d'écriture décimale
214 static double calcule_tic(float tic_min)
215 {
216  double res = 1;
217  double pnorm = std::pow(10.0, (double) std::floor(std::log10(tic_min)));
218 
219  float v = tic_min / pnorm;
220  // 0.15 -> 1.5 -> tic = 0.2
221  // 23.15 -> 2.31 -> tic = 25
222 
223 
224 
225  if(v == 1)
226  {
227  // Dans ce cas, tic_min est déjà une puissance de 10
228  res = 1;
229  }
230  else if(v <= 2)
231  res = 2;
232  else if(v <= 2.5)
233  res = 2.5;
234  else if(v <= 5.0)
235  res = 5.0;
236  else
237  res = 10.0;
238 
239  res *= pnorm;
240 
241  //msg("tic min = {} -> tic = {} (pnorm = {})", tic_min, res, pnorm);
242 
243  return res;
244 }

717  vmin = log10(vmin);
718  vmax = log10(vmax);

721  DBG(msg("Axe [{}]: dim={}, cfg.min={}, cfg.max={}, pixels_min={}, ecart-pixel={}, mode={}",

723  config.grille_majeure.pixels_par_tic_min,
724  axe.config.ecart_pixel,

732  // Calcul des tics majeurs et mineurs
733 
734  // δ = diff vmax - vmin, en unité utilisateur
735  float δ = std::abs(vmax - vmin);
736 
737  // exemple : δ = 15
738  // => unité = 1
739 
740  // => On cherche le plus petit tic majeur < au plus petit
741  // de abs(vmax) / abs(vmin)
742 
743  float pixels_par_tic_majeur_min = config.grille_majeure.pixels_par_tic_min;
744 
745  essai:
746 
747  // Imposé par la résolution
748  // eg lsb_par_pixel / pixel_par_tic_majeur_min
749 
750  // Exemple : δ = 1000, axe.dim = 100, pptm_min = 50
751  // tic_majeur_min = 10 * 50 = 500 LSB
752 
753  // Tic majeur mini, en unité LSB (ou LSB_LOG en mode logarithmique)
754  // Au moins ... pixels => au moins ... en valeur utilisateur
755  float tic_majeur_min = (δ / axe.dim) * pixels_par_tic_majeur_min;
756  float tic_mineur_min = (δ / axe.dim) * config.grille_mineure.pixels_par_tic_min;
757 
758 
759 
760 
761  tic_majeur = calcule_tic(tic_majeur_min);
762 
763  DBG(msg(" δ={}, tcmin={} -> essai tic majeur = {}", δ, tic_majeur_min, tic_majeur);)
764 
765  // En mode logarithmique, travaille au mimimun par puissance entière de 10
766  if(mode_log)
767  tic_majeur = std::ceil(tic_majeur);
768 
769  if(axe.config.ecart_pixel != -1)
770  tic_majeur = axe.config.ecart_pixel * (δ / axe.dim);
771

773  {
774  msg_avert("Tic majeur : {}", tic_majeur);
775  return;
776  }
777 
778  if(tic_majeur <= std::numeric_limits<double>::epsilon())
779  {
780  tic_majeur = 10 * std::numeric_limits<double>::epsilon();

782  }
783 
784  axe.ecart_tics_majeurs_valeur = tic_majeur;
785  axe.ecart_tics_majeurs_pixels = tic_majeur * (axe.dim / δ);
786 
787  DBG(msg(" Essai tic majeur = {}, pp_par_tic_majeur_min = {} -> ecart_tic_majeurs_pix = {}...", tic_majeur, pixels_par_tic_majeur_min, axe.ecart_tics_majeurs_pixels);)
788 
789  tic_mineur = tic_majeur / config.grille_majeure.sous_graduation;
790  axe.tics_mineurs_presents = tic_mineur >= tic_mineur_min;
791 
792  axe.tics_majeurs_pos.clear();
793  axe.tics_mineurs_pos.clear();

795  // Entre tic_majeur_min et δ, on cherche des valeurs
796  // pas trop bêtes
797 
798  bool zero_present = (vmin <= 0) && (vmax >= 0);

800  // Les variables tk, tic_majeur, etc. sont en logs
801  auto adapt = [&](double v)
802  {
803  if(mode_log)
804  return std::pow(10.0f, v);
805  return v;
806  };

808  // Ajoute des tics mineurs au dessus de tk (tic majeur)s
809  auto ajouter_tics_mineurs = [&](float tk)
810  {
811  if(!axe.tics_mineurs_presents)
812  return;
813 
814  // En mode logarithmique :
815  // tic_mineur, tic_majeur sont en logarithmique
816  // d'ailleurs, tic_mineur n'a pas de sens
817  // ( tic_mineur = tic_majeur / config.grille_majeure.sous_graduation;)
818  // On commence par tk
819  // tk est en logarithmique.
820  // Par exemple tk = -10
821  // disons tic_majeur = 10
822  // On va y aller de manière linéaire et pas log (pas de 2 en 2 en log)
823  // 10^tk, 10^tk - R, 10^tk - 2*R, ... , [10^(tk-ticmajeur)
824  // => en log : log(10^tk - R), log(10^tk - 2*R), etc...
825  // R = ?
826  // Exemple : tic_majeur = 5, 5 pas
827  // Intervalle linéaire : 10^5
828  // à couper en 5 parties :
829  // 0, [10^5/5, 2*10^5/5, 3*10^5/5, 4*10^5/5,] 5*10^5/5
830  // => R = 10^(tic_majeur) / nb_grad
831 
832  if(mode_log)
833  {
834  // si sous grad = 1/10:
835  // R = 10^(tk+M) / 10 = 10^{tk+M-1}
836  // tk0 = log10(10^tk + 10^{tk+M-1})
837  // log10(10^tk * (1 + 10^{M-1})
838  // tk + log10(1 + 10^{M-1})
839  // ~ tk + M - 1
840  // si M = 5
841  //
842  // Exemple : tic_majeur = 1, tk = 1 (10^1 -> 10^2)
843  // R = 100 / 10 = 10;
844  // tkm0 = log10(20) = 1.3
845 
846

865  axe.tics_mineurs_pos.push_back(adapt(tkm));

868  }
869  };

875  double tk = 0;//tic_majeur;

881  axe.tics_majeurs_pos.push_back(adapt(-tk));

893

895  if(vmin >= 0)
896  {
897  // => Le plus petit multiple de tic_majeur > config.min
898  DBG(msg("vmin={}, tmaj={} --> ceil={}", vmin, tic_majeur, tic_majeur * std::ceil(vmin / tic_majeur));)
899  tk = tic_majeur * std::ceil(vmin / tic_majeur);
900 
901  if(!axe.tic_ok(adapt(tk)))
902  tk += tic_majeur;
903 
904  DBG(msg(" tk={}, tic ok = {}", tk, axe.tic_ok(adapt(tk)));)
905  DBG(msg("valeur_vers_pixel : {}", axe.valeur_vers_pixel(tk));)
906 
907  while(axe.tic_ok_souple(adapt(tk)))
908  {
909  if(axe.tic_ok(adapt(tk)))
910  axe.tics_majeurs_pos.push_back(adapt(tk));
911  ajouter_tics_mineurs(tk);
912  tk += tic_majeur;
913  }
914  }
915  // vmin < vmax < 0
916  else
917  {
918  // => Le plus grand multiple de tic_majeur < vmax
919  tk = -tic_majeur * std::ceil(-vmax / tic_majeur);
920

924  {
925  if(axe.tic_ok(adapt(tk)))
926  axe.tics_majeurs_pos.push_back(adapt(tk));
927  ajouter_tics_mineurs(tk);
928  tk -= tic_majeur;
929  }
930  }
